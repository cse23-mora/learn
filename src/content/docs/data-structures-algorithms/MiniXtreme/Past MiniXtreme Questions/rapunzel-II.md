```cpp
#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

bool cmp(const pair<int, int>& a, const pair<int, int>& b) {
    return a.first < b.first;
}

int main() {
    int tests;
    cin >> tests;
    
    for(int t = 0; t < tests; t++){
        int n;
        cin >> n;
        
        vector<pair<int,int>> hps(n);
        for (int j = 0; j < n; ++j) {
            int k;
            cin >> k;
            hps[j] = {k, j};
            
        }
        
        vector<pair<int,int>> sortedhps = hps;
        
        sort(sortedhps.begin(), sortedhps.end(),cmp);
        
        for(pair<int,int> h: hps){
            int req = 2*h.first;
            int leader = -1;

            auto it = upper_bound(sortedhps.begin(), sortedhps.end(), make_pair(req,0), cmp);
            
            while (it != sortedhps.begin()) {
                --it;
                if (it->second != h.second) {
                    leader = it -> first;
                    break;
                }
            }

            cout << leader << " ";
        }
        cout << "\n";
        
    }
    
    return 0;
}
```
